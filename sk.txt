local function pushRemoteLog(name, detail)
	table.insert(remoteLogEntries, 1, string.format("[%.1f] %s: %s", os.clock() % 1000, name, detail))
	while #remoteLogEntries > REMOTE_LOG_MAX do
		table.remove(remoteLogEntries)
	end
	if remoteLogLabel then
		remoteLogLabel.Text = table.concat(remoteLogEntries, "\n")
	end
end

local function getRemote(name)
	local remotes = ReplicatedStorage:FindFirstChild("Remotes")
	return remotes and remotes:FindFirstChild(name)
end

local function fireRemote(name, ...)
	local remote = getRemote(name)
	if not remote then
		pushRemoteLog(name, "not found")
		if statusLabel then
			statusLabel.Text = name .. ": not found"
		end
		return false, "not found"
	end
	local args = { ... }
	local ok, err = pcall(function()
		if remote:IsA("RemoteEvent") then
			remote:FireServer(table.unpack(args))
		elseif remote:IsA("RemoteFunction") then
			remote:InvokeServer(table.unpack(args))
		end
	end)
	local detail = ok and "ok" or tostring(err)
	pushRemoteLog(name, detail)
	if statusLabel then
		statusLabel.Text = name .. ": " .. detail
	end
	return ok, detail
end

local function findWinBlock(stageNum)
	local structure = workspace:FindFirstChild("Structure")
	if not structure then
		return nil
	end
	local stage = structure:FindFirstChild("Stage" .. stageNum)
	if not stage then
		return nil
	end
	for _, desc in stage:GetDescendants() do
		if desc:IsA("BasePart") and string.find(desc.Name, "WinBlock", 1, true) then
			return desc
		end
	end
	return nil
end

local function findTreadmillPart(tagName)
	for _, inst in CollectionService:GetTagged(tagName) do
		if inst:IsA("BasePart") then
			return inst
		end
		if inst:IsA("Model") then
			local part = inst.PrimaryPart or inst:FindFirstChildWhichIsA("BasePart", true)
			if part then
				return part
			end
		end
	end
	return nil
end

local function makeSlider(parent, labelText, minVal, maxVal, defaultVal, yPos, onChange)
	local label = Instance.new("TextLabel")
	label.Size = UDim2.new(1, -20, 0, 20)
	label.Position = UDim2.new(0, 10, 0, yPos)
	label.BackgroundTransparency = 1
	label.TextColor3 = Color3.fromRGB(200, 200, 200)
	label.TextSize = 13
	label.Font = Enum.Font.Gotham
	label.TextXAlignment = Enum.TextXAlignment.Left
	label.Text = labelText .. ": " .. defaultVal
	label.Parent = parent

	local track = Instance.new("Frame")
	track.Size = UDim2.new(1, -20, 0, 10)
	track.Position = UDim2.new(0, 10, 0, yPos + 24)
	track.BackgroundColor3 = Color3.fromRGB(60, 60, 60)
	track.BorderSizePixel = 0
	track.Parent = parent
	makeCorner(track, 5)

	local fill = Instance.new("Frame")
	fill.Size = UDim2.new((defaultVal - minVal) / (maxVal - minVal), 0, 1, 0)
	fill.BackgroundColor3 = Color3.fromRGB(80, 180, 255)
	fill.BorderSizePixel = 0
	fill.Parent = track
	makeCorner(fill, 5)

	local knob = Instance.new("TextButton")
	knob.Size = UDim2.new(0, 18, 0, 18)
	knob.Position = UDim2.new((defaultVal - minVal) / (maxVal - minVal), -9, 0.5, -9)
	knob.BackgroundColor3 = Color3.fromRGB(255, 255, 255)
	knob.BorderSizePixel = 0
	knob.Text = ""
	knob.Parent = track
	makeCorner(knob, 9)

	local dragging = false
	knob.MouseButton1Down:Connect(function()
		dragging = true
	end)
	UserInputService.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = false
		end
	end)
	UserInputService.InputChanged:Connect(function(input)
		if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
			local ratio = math.clamp((input.Position.X - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)
			local value
			if maxVal <= 1 and minVal < 1 then
				value = math.floor((minVal + ratio * (maxVal - minVal)) * 100) / 100
			else
				value = math.floor(minVal + ratio * (maxVal - minVal))
			end
			fill.Size = UDim2.new(ratio, 0, 1, 0)
			knob.Position = UDim2.new(ratio, -9, 0.5, -9)
			label.Text = labelText .. ": " .. value
			onChange(value)
		end
	end)
	return label
end

local function ensureWalkLocomotion()
	if flyConnection then
		flyConnection:Disconnect()
		flyConnection = nil
	end
	if bodyVelocity then
		bodyVelocity:Destroy()
		bodyVelocity = nil
	end
	if bodyGyro then
		bodyGyro:Destroy()

	end

		humanoid.PlatformStand = false

		flying = false
		flyBtn.Text = "Fly: OFF"
		flyBtn.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
	end
end

local function disconnectSpeedFarmMove()
	pcall(function()
		RunService:UnbindFromRenderStep(SPEED_FARM_BIND)
	end)
	if speedFarmConnection then
		speedFarmConnection:Disconnect()
		speedFarmConnection = nil
	end
end

local function disconnectTreadmillMove()
	if treadmillMoveConnection then
		treadmillMoveConnection:Disconnect()
		treadmillMoveConnection = nil
	end
end

local function getMoveStepEvent()
	return RunService.PreSimulation or RunService.Heartbeat
end

local function startFarming()
	farmGeneration += 1
	local gen = farmGeneration

	local function glideTo(targetPos, duration, level)
		if not farmingEnabled then
			return
		end
		local startPos = rootPart.Position
		local steps = 60
		local stepTime = duration / steps
		for i = 1, steps do
			if not farmingEnabled then
				break
			end
			local alpha = i / steps
			rootPart.CFrame = CFrame.new(startPos:Lerp(targetPos, alpha))
			farmInfoLabel.Text = string.format(
				"Level %d\nMoving to X:%.1f Y:%.1f Z:%.1f",
				level,
				targetPos.X,
				targetPos.Y,
				targetPos.Z
			)
			task.wait(stepTime)
		end
	end

	if selectedLevel == 2 then
		local speedFarmPos = FARM_LOCATIONS[2].start
		local nextTeleport = 0
		ensureWalkLocomotion()
		disconnectSpeedFarmMove()
		RunService:BindToRenderStep(SPEED_FARM_BIND, Enum.RenderPriority.Character.Value + 1, function()
			if not farmingEnabled or gen ~= farmGeneration or selectedLevel ~= 2 then
				return
			end
			if not humanoid or not rootPart or not humanoid.Parent or humanoid.Health <= 0 then
				return
			end

			ensureWalkLocomotion()

			-- Snap back to farm spot every 1s (fixed X/Y/Z from config)
			local now = os.clock()
			if now >= nextTeleport then
				local standPos = Vector3.new(speedFarmPos.X, speedFarmPos.Y, speedFarmPos.Z)
				rootPart.CFrame = CFrame.lookAt(standPos, standPos + Vector3.new(0, 0, 1))
				nextTeleport = now + 1
				farmInfoLabel.Text = string.format(
					"Speed Farm\nAt X:%.0f Y:%.0f Z:%.0f (+Z walk)",
					standPos.X,
					standPos.Y,
					standPos.Z
				)
			end

			-- +Z walk input every frame (between teleports too)
			humanoid.WalkSpeed = WALK_SPEED
			humanoid:Move(Vector3.new(0, 0, 1), false)
		end)
		speedFarmConnection = true
		return
	end

	disconnectSpeedFarmMove()

	task.spawn(function()
		while farmingEnabled and gen == farmGeneration do
			if not humanoid or not rootPart or not humanoid.Parent then
				task.wait(0.1)
				continue
			end
			local data = FARM_LOCATIONS[selectedLevel]
			rootPart.CFrame = CFrame.new(data.start)
			farmInfoLabel.Text = "Level 1\nTeleporting to start..."
			task.wait(0.1)
			for _, wp in ipairs(data.waypoints) do
				if not farmingEnabled or gen ~= farmGeneration then
					break
				end
				glideTo(wp, 1, selectedLevel)
			end
			farmInfoLabel.Text = "Level 1\nWaiting at end..."
			task.wait(1)
		end
		if gen == farmGeneration and not farmingEnabled then
			farmInfoLabel.Text = "Farming inactive"
		end
	end)
end

local function stopFarming()
	farmingEnabled = false
	farmGeneration += 1
	disconnectSpeedFarmMove()
	if humanoid and humanoid.Parent then
		humanoid:Move(Vector3.zero, false)
	end
	farmInfoLabel.Text = "Farming inactive"
end

local function getRebirthBtn(): GuiButton?
	local playerGui = player:FindFirstChild("PlayerGui")
	if not playerGui then
		return nil
	end
	for _, inst in CollectionService:GetTagged("RebirthConfirmBtn") do
		if inst:IsDescendantOf(playerGui) and inst:IsA("GuiButton") then
			return inst
		end
	end
	local current: Instance? = playerGui
	for _, name in { "SpeedGameUI", "Modals", "RebirthModal", "RebirthFrame", "RebirthBtn" } do
		current = current and current:FindFirstChild(name)
	end
	if current and current:IsA("GuiButton") then
		return current
	end
	return nil
end

local function triggerRebirth(): (boolean, string)
	local rebirthBtn = getRebirthBtn()
	if rebirthBtn then
		pcall(function()
			rebirthBtn:Activate()
		end)
	end
	local ok = fireRemote("Rebirth")
	return ok, ok and "Rebirth fired" or "Rebirth failed"
end

local function startRebirthFarm()

		end
		if gen == rebirthGeneration and not rebirthFarmEnabled then
			rebirthInfoLabel.Text = "Rebirth farm inactive"
		end
	end)
end

local function stopRebirthFarm()

local function startTreadmillFarm()
	treadmillGeneration += 1
	local gen = treadmillGeneration
	local remoteAccumulator = 0
	local placedOnTreadmill = false
	disconnectTreadmillMove()
	treadmillMoveConnection = getMoveStepEvent():Connect(function(dt)
		if not treadmillFarmEnabled or gen ~= treadmillGeneration then
			return
		end
		if not humanoid or not rootPart or not humanoid.Parent or humanoid.Health <= 0 then
			return
		end

		local entry = TREADMILL_TYPES[treadmillTypeIndex]
		if not entry then
			return
		end

		if not placedOnTreadmill then
			local part = findTreadmillPart(entry.tag)
			if part then
				local standPos = part.Position + Vector3.new(0, 4, 0)
				rootPart.CFrame = CFrame.lookAt(standPos, standPos + Vector3.new(0, 0, -1))
				placedOnTreadmill = true
			end
		end

		if humanoid.PlatformStand then
			humanoid.PlatformStand = false
		end
		humanoid.WalkSpeed = WALK_SPEED
		humanoid:Move(Vector3.new(0, 0, -1), false)

		remoteAccumulator += dt
		if remoteAccumulator >= 0.15 then
			remoteAccumulator = 0
			fireRemote("TreadmillSignal", true)
			fireRemote("UpdateSpeed", entry.remote)
		end
	end)
end

local function startFly()
	bodyVelocity = Instance.new("BodyVelocity")
	bodyVelocity.Velocity = Vector3.zero
	bodyVelocity.MaxForce = Vector3.new(1e5, 1e5, 1e5)
	bodyVelocity.Parent = rootPart

	bodyGyro = Instance.new("BodyGyro")
	bodyGyro.MaxTorque = Vector3.new(1e5, 1e5, 1e5)
	bodyGyro.D = 100
	bodyGyro.Parent = rootPart

	flyConnection = RunService.Heartbeat:Connect(function()
		if not flying then
			return
		end
		local camera = workspace.CurrentCamera
		local direction = Vector3.zero
		if UserInputService:IsKeyDown(Enum.KeyCode.W) then
			direction += camera.CFrame.LookVector
		end
		if UserInputService:IsKeyDown(Enum.KeyCode.S) then
			direction -= camera.CFrame.LookVector
		end
		if UserInputService:IsKeyDown(Enum.KeyCode.A) then
			direction -= camera.CFrame.RightVector
		end
		if UserInputService:IsKeyDown(Enum.KeyCode.D) then
			direction += camera.CFrame.RightVector
		end
		if UserInputService:IsKeyDown(Enum.KeyCode.Space) then
			direction += Vector3.new(0, 1, 0)
		end
		if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then
			direction -= Vector3.new(0, 1, 0)
		end
		bodyVelocity.Velocity = direction.Magnitude > 0 and direction.Unit * FLY_SPEED or Vector3.zero
		bodyGyro.CFrame = camera.CFrame
	end)

	humanoid.PlatformStand = true
end

local function stopFly()
	if flyConnection then
		flyConnection:Disconnect()
		flyConnection = nil
	end
	if bodyVelocity then
		bodyVelocity:Destroy()
		bodyVelocity = nil
	end
	if bodyGyro then
		bodyGyro:Destroy()
		bodyGyro = nil
	end
	if humanoid and humanoid.Parent then
		humanoid.PlatformStand = false
	end
end
